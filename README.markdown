# Installing

    easy_install pbundler
    pbundle

or

    virtualenv venv-camel
    . ./venv-camel/bin/activate
    pip install -r requirements.txt

# Configuration

    cp config.json.example config.json
    editor config.json

# Running (Debug mode)

    pbundle-py debug.py

or

    . ./venv-camel/bin/activate
    python debug.py
